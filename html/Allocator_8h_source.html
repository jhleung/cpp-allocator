153  pointer allocate (size_type n) {
154  if ((int)n < 0) {
155  bad_alloc exception;
156  throw exception;
157  }
158 
159  for (int i = 0; i < N - 1; i++) {
160  int blockSize = (*this)[i];
161  if (blockSize > 0 && blockSize < N) {
162  if ((*this)[i+blockSize+4] == blockSize) {
163  int dataSize = n * sizeof(T);
164  int blockSizeNeeded = dataSize + 2 * sizeof(int);
165  int allocatedBlockSize = blockSizeNeeded;
166 
167  //entire array is free and allocation requests entire array
168  if (blockSize == blockSizeNeeded) {
169  (*this)[i] = -1 * (n * sizeof(T)) - 8;
170  (*this)[i+blockSize+4] = -1 * (n * sizeof(T)) - 8;
171  pointer ptr = reinterpret_cast<pointer>(&a[i+4]);
172  return ptr;
173  }
174 
175  //blocksize+8 to account for sentinels
176  if (blockSize + 8 >= blockSizeNeeded) {
177  //found an exact fit
178  if (blockSize + 8 - blockSizeNeeded == 0) {
179  (*this)[i] = -1 * (n * sizeof(T));
180  (*this)[i+blockSize+4] = -1 * (n * sizeof(T));
181  pointer ptr = reinterpret_cast<pointer>(&a[i+4]);
182  return ptr;
183  }
184  if (blockSize + 8 - blockSizeNeeded < sizeof(T) + 2 * sizeof(int)){
185  blockSizeNeeded = blockSize;
186  }
187 
188  (*this)[i] = -1 * (n * sizeof(T));
189  (*this)[i+4+dataSize] = -1 * dataSize;
190 
191  //set new sentinels for new free block
192  (*this)[i+4+dataSize+4] = blockSize - blockSizeNeeded;
193  (*this)[i+blockSize+4] = blockSize - blockSizeNeeded;
194 
195  pointer ptr = reinterpret_cast<pointer>(&a[i+4]);
196  return ptr;
197  }
198  }
199  }
200  if (blockSize < 0 && -1 * blockSize < N && (*this)[i+(-1*blockSize)+4] == blockSize) {
201  //skip over these
202  i += (-1*blockSize) + 8 - 1;
203  }
204  }
205  bad_alloc exception;

231  void deallocate (pointer p, size_type) {
232  // change sentinels from negative to positive
233  // coalesce:
234  // both adj blocks are free
235  // left adj block is free
236  // right adj block is free
237  int *pp = reinterpret_cast<int*>(p);
238  int sentinelAPos = -1;
239  int& sentinelA = *(pp+sentinelAPos);
240  int sentinelBPos = (-1*(sentinelA))/4;
241  int& sentinelB = *(pp+sentinelBPos);
242 
243  // is ptr given valid?
244  if (sentinelA < 0 && sentinelA == sentinelB) {
245  sentinelA *= -1;
246  sentinelB *= -1;
247 
248  //check right adjacent block to deallocated block's sentinels
249  int sentinelRAdjAPos = sentinelBPos + 1;
250 
251  //check left adj block...
252  int sentinelLAdjBPos = sentinelAPos - 1;
253 
254  // both adj blocks are also free after deallocation of requested block
255  if (pp+sentinelRAdjAPos <= &(*this)[N-1] && pp+sentinelLAdjBPos >= &(*this)[0]) {
256  int& sentinelRAdjA = *(pp + sentinelRAdjAPos);
257  int sentinelRAdjBPos = (abs(sentinelRAdjA))/4 + sentinelRAdjAPos + 1;
258  int& sentinelRAdjB = *(pp + sentinelRAdjBPos);
259 
260  int& sentinelLAdjB = *(pp + sentinelLAdjBPos);
261  int sentinelLAdjAPos = sentinelLAdjBPos - abs(sentinelLAdjB)/4 - 1;
262  int& sentinelLAdjA = *(pp + sentinelLAdjAPos);
263 
264  if (sentinelRAdjA > 0 && sentinelRAdjA == sentinelRAdjB && sentinelLAdjA > 0 && sentinelLAdjA == sentinelLAdjB) {
265  int coalesceSize = sentinelLAdjA + sentinelA + sentinelRAdjA + 16;
266  sentinelLAdjA = coalesceSize;
267  sentinelRAdjB = coalesceSize;
268 
269  //clear inner sentinels
270  sentinelLAdjB = 0;
271  sentinelA = 0;
272  sentinelB = 0;
273  sentinelRAdjA = 0;
274  }
275  }
276  // only right adj block is also free
277  if (pp+sentinelRAdjAPos <= &(*this)[N-1]) {
278  int& sentinelRAdjA = *(pp + sentinelRAdjAPos);
279  int sentinelRAdjBPos = (abs(sentinelRAdjA))/4 + sentinelRAdjAPos + 1;
280  int& sentinelRAdjB = *(pp + sentinelRAdjBPos);
281 
282  if (sentinelRAdjA > 0 && sentinelRAdjA == sentinelRAdjB) {
283  int coalesceSize = sentinelA + sentinelRAdjA + 8;
284  sentinelA = coalesceSize;
285  sentinelRAdjB = coalesceSize;
286 
287  //clear inner sentinels
288  sentinelB = 0;
289  sentinelRAdjA = 0;
290  }
291  }
292  if(pp+sentinelLAdjBPos >= &(*this)[0]) {
293  // only left adj block is also free
294  int& sentinelLAdjB = *(pp + sentinelLAdjBPos);
295  int sentinelLAdjAPos = sentinelLAdjBPos - abs(sentinelLAdjB)/4 - 1;
296  int& sentinelLAdjA = *(pp + sentinelLAdjAPos);
297  if (sentinelLAdjA > 0 && sentinelLAdjA == sentinelLAdjB) {
298  int coalesceSize = sentinelB + sentinelLAdjA + 8;
299  sentinelB = coalesceSize;
300  sentinelLAdjA = coalesceSize;

304  sentinelLAdjB = 0;
305  }
